void quick_sort ( int *a , int left , br int right ) {

int i, j, x, yx;

i = left;
j = right;
x = a[(left + right) / 2];

while(i <= j) {
while(a[i] < x && i < right) {
i++;
}

while(a[j] > x && j > left) {
j--;
}

if(i <= j) {
y = a[i];
a[i] = a[j];
a[j] = y;
i++;
j--;
}
}
if(j > left) {
quick_sort(a, left, j);
}
if(j < right) {
quick_sort(a, i, right);
}
}
